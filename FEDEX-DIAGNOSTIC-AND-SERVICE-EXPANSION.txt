```typescript
/**
 * Shipping API Diagnostic Test
 * Run with: npx tsx scripts/test-shipping.ts
 * DELETE THIS FILE after running.
 */
import * as dotenv from "dotenv";
dotenv.config();

const SHIPPO_TOKEN = process.env.SHIPPO_API_KEY || process.env.SHIPPO_API_TOKEN || "";
const FEDEX_API_KEY = process.env.FEDEX_PARCEL_API_KEY || "";
const FEDEX_SECRET = process.env.FEDEX_PARCEL_SECRET_KEY || "";
const FEDEX_URL = process.env.FEDEX_PARCEL_URL || "https://apis-sandbox.fedex.com";
const FEDEX_ACCOUNT = process.env.FEDEX_ACCOUNT_NUMBER || "";

async function testShippo() {
  console.log("=== SHIPPO DIAGNOSTIC ===");
  console.log("Token:", SHIPPO_TOKEN ? `SET (${SHIPPO_TOKEN.length} chars, starts with ${SHIPPO_TOKEN.slice(0, 12)}...)` : "MISSING");

  if (!SHIPPO_TOKEN) {
    console.log("SKIP — no Shippo token configured");
    return;
  }

  try {
    const res = await fetch("https://api.goshippo.com/shipments", {
      method: "POST",
      headers: {
        Authorization: `ShippoToken ${SHIPPO_TOKEN}`,
        "Content-Type": "application/json",
      },
      body: JSON.stringify({
        address_from: {
          street1: "123 Main St",
          city: "Waterville",
          state: "ME",
          zip: "04901",
          country: "US",
        },
        address_to: {
          street1: "456 Buyer St",
          city: "New York",
          state: "NY",
          zip: "10001",
          country: "US",
        },
        parcels: [{
          length: "12",
          width: "10",
          height: "6",
          distance_unit: "in",
          weight: "5",
          mass_unit: "lb",
        }],
        async: false,
      }),
    });

    console.log("Shippo status:", res.status);

    if (!res.ok) {
      const errText = await res.text();
      console.log("Shippo FAILED:", errText.slice(0, 500));
      return;
    }

    const data = await res.json();
    const rates = data.rates || [];
    console.log(`Shippo returned ${rates.length} total rates`);
    console.log("");

    // Group by carrier
    const byCarrier: Record<string, any[]> = {};
    for (const r of rates) {
      const carrier = r.provider || "Unknown";
      if (!byCarrier[carrier]) byCarrier[carrier] = [];
      byCarrier[carrier].push(r);
    }

    for (const [carrier, carrierRates] of Object.entries(byCarrier)) {
      console.log(`--- ${carrier} (${carrierRates.length} services) ---`);
      for (const r of carrierRates.sort((a: any, b: any) => parseFloat(a.amount) - parseFloat(b.amount))) {
        const service = r.servicelevel?.name || r.servicelevel?.token || "Unknown";
        console.log(`  $${parseFloat(r.amount).toFixed(2)} — ${service} (${r.estimated_days || "?"} days) [token: ${r.servicelevel?.token || "n/a"}]`);
      }
    }

    // Show any carriers NOT in our ALLOWED list
    const allowed = ["usps", "ups", "fedex"];
    const excluded = rates.filter((r: any) => !allowed.includes(r.provider?.toLowerCase()));
    if (excluded.length > 0) {
      console.log("");
      console.log(`--- EXCLUDED carriers (${excluded.length} rates not shown to users) ---`);
      for (const r of excluded) {
        console.log(`  ${r.provider} — ${r.servicelevel?.name} $${r.amount}`);
      }
    }
  } catch (e) {
    console.log("Shippo EXCEPTION:", e);
  }
  console.log("");
}

async function testFedEx() {
  console.log("=== FEDEX DIRECT API DIAGNOSTIC ===");
  console.log("API Key:", FEDEX_API_KEY ? `SET (${FEDEX_API_KEY.length} chars)` : "MISSING");
  console.log("Secret:", FEDEX_SECRET ? `SET (${FEDEX_SECRET.length} chars)` : "MISSING");
  console.log("URL:", FEDEX_URL);
  console.log("Account:", FEDEX_ACCOUNT || "MISSING");

  if (!FEDEX_API_KEY || !FEDEX_SECRET) {
    console.log("SKIP — no FedEx credentials");
    return;
  }

  // Auth
  let token = "";
  try {
    const authRes = await fetch(`${FEDEX_URL}/oauth/token`, {
      method: "POST",
      headers: { "Content-Type": "application/x-www-form-urlencoded" },
      body: new URLSearchParams({
        grant_type: "client_credentials",
        client_id: FEDEX_API_KEY,
        client_secret: FEDEX_SECRET,
      }),
    });
    if (authRes.ok) {
      const authData = await authRes.json();
      token = authData.access_token;
      console.log("Auth: SUCCESS (token length:", token.length + ")");
    } else {
      const body = await authRes.text();
      console.log("Auth: FAILED — status:", authRes.status, "body:", body.slice(0, 300));
      return;
    }
  } catch (e) {
    console.log("Auth EXCEPTION:", e);
    return;
  }

  // Rate request
  const reqBody = {
    accountNumber: { value: FEDEX_ACCOUNT },
    requestedShipment: {
      shipper: { address: { postalCode: "04901", countryCode: "US" } },
      recipient: { address: { postalCode: "10001", countryCode: "US", residential: true } },
      pickupType: "DROPOFF_AT_FEDEX_LOCATION",
      rateRequestType: ["LIST", "ACCOUNT"],
      shippingChargesPayment: {
        paymentType: "SENDER",
        payor: {
          responsibleParty: {
            accountNumber: { value: FEDEX_ACCOUNT },
          },
        },
      },
      requestedPackageLineItems: [{
        weight: { units: "LB", value: 5 },
        dimensions: { length: 12, width: 10, height: 6, units: "IN" },
      }],
    },
  };

  try {
    const rateRes = await fetch(`${FEDEX_URL}/rate/v1/rates/quotes`, {
      method: "POST",
      headers: {
        Authorization: `Bearer ${token}`,
        "Content-Type": "application/json",
        "X-locale": "en_US",
      },
      body: JSON.stringify(reqBody),
    });

    const rateBody = await rateRes.text();
    console.log("Rate status:", rateRes.status);

    if (rateRes.ok) {
      const rateData = JSON.parse(rateBody);
      const details = rateData.output?.rateReplyDetails || [];
      console.log(`Rate: SUCCESS — ${details.length} services returned`);
      for (const d of details) {
        const charge = d.ratedShipmentDetails?.[0]?.totalNetCharge ?? d.ratedShipmentDetails?.[0]?.totalNetFedExCharge ?? 0;
        const days = d.commit?.dateDetail?.dayCount || d.commit?.transitDays?.amount || "?";
        console.log(`  $${charge} — ${d.serviceType} (${days} days)`);
      }
    } else {
      console.log("Rate: FAILED");
      console.log("FULL ERROR RESPONSE:");
      console.log(rateBody);
    }
  } catch (e) {
    console.log("Rate EXCEPTION:", e);
  }
  console.log("");
}

async function main() {
  console.log("========================================");
  console.log("LEGACYLOOP SHIPPING DIAGNOSTIC");
  console.log("Test route: Waterville ME 04901 → New York NY 10001");
  console.log("Test parcel: 5 lbs, 12×10×6 in");
  console.log("========================================");
  console.log("");

  await testShippo();
  await testFedEx();

  console.log("========================================");
  console.log("DIAGNOSTIC COMPLETE");
  console.log("========================================");
}

main();
```
