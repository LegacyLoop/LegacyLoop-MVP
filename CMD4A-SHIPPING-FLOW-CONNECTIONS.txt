// ─── Carrier ID Cache ───────────────────────────────────────────────────────
let cachedCarrierIds: string[] | null = null;
let carrierCacheTime = 0;
const CARRIER_CACHE_TTL = 5 * 60 * 1000; // 5 minutes

async function getConnectedCarrierIds(): Promise<string[]> {
  if (cachedCarrierIds && Date.now() - carrierCacheTime < CARRIER_CACHE_TTL) {
    return cachedCarrierIds;
  }
  try {
    const carriers = await listCarriers();
    const ids = carriers.map((c: any) => c.carrier_id).filter(Boolean);
    if (ids.length > 0) {
      cachedCarrierIds = ids;
      carrierCacheTime = Date.now();
      console.log(`[ShipEngine] Connected carriers cached: ${ids.length} carriers — ${carriers.map((c: any) => c.friendly_name || c.carrier_id).join(", ")}`);
    }
    return ids;
  } catch {
    return [];
  }
}

  const getTrackingUrl = (carrier: string, tracking: string): string | null => {
    const c = (carrier || "").toLowerCase();
    if (c.includes("usps")) return `https://tools.usps.com/go/TrackConfirmAction?tLabels=${tracking}`;
    if (c.includes("ups")) return `https://www.ups.com/track?tracknum=${tracking}`;
    if (c.includes("fedex")) return `https://www.fedex.com/fedextrack/?trknbr=${tracking}`;
    if (c.includes("dhl")) return `https://www.dhl.com/us-en/home/tracking.html?tracking-id=${tracking}`;
    return null;
  };

```typescript
/**
 * ShipEngine Carrier Discovery Diagnostic
 * Run: npx tsx scripts/test-shipengine-carriers.ts
 * DELETE after running.
 */
require("dotenv").config();

const KEY = process.env.SHIPENGINE_API_KEY || "";
const BASE = process.env.SHIPENGINE_BASE_URL || "https://api.shipengine.com";

async function main() {
  console.log("========================================");
  console.log("SHIPENGINE CARRIER DISCOVERY");
  console.log("========================================\n");
  console.log("API Key:", KEY ? `SET (${KEY.length} chars, starts with ${KEY.slice(0, 10)}...)` : "MISSING");
  console.log("Base URL:", BASE);

  if (!KEY) {
    console.log("\nSKIP — no ShipEngine key configured");
    return;
  }

  // List carriers
  console.log("\n--- Connected Carriers ---");
  try {
    const res = await fetch(`${BASE}/v1/carriers`, {
      headers: { "API-Key": KEY, "Content-Type": "application/json" },
    });
    console.log("Status:", res.status);
    if (!res.ok) {
      console.log("FAILED:", (await res.text()).slice(0, 300));
      return;
    }
    const data = await res.json();
    const carriers = data.carriers || [];
    console.log(`Total carriers: ${carriers.length}`);
    for (const c of carriers) {
      console.log(`  ${c.carrier_id} — ${c.friendly_name || c.nickname || "unnamed"} (${c.carrier_code || "no-code"})`);
      if (c.services) {
        for (const s of c.services.slice(0, 5)) {
          console.log(`    → ${s.service_code}: ${s.name}`);
        }
        if (c.services.length > 5) console.log(`    → ... and ${c.services.length - 5} more services`);
      }
    }

    if (carriers.length === 0) {
      console.log("⚠️  0 carriers connected. Add carriers in ShipEngine dashboard.");
      console.log("   For sandbox: use carrier_id se-1115747");
    }

    // Try a rate estimate with discovered carriers
    if (carriers.length > 0) {
      console.log("\n--- LTL Rate Estimate Test ---");
      const carrierIds = carriers.map((c: any) => c.carrier_id);
      console.log("Using carrier IDs:", carrierIds.join(", "));

      const rateRes = await fetch(`${BASE}/v1/rates/estimate`, {
        method: "POST",
        headers: { "API-Key": KEY, "Content-Type": "application/json" },
        body: JSON.stringify({
          rate_options: { carrier_ids: carrierIds },
          shipment: {
            ship_from: { name: "LegacyLoop", address_line1: "123 Main St", city_locality: "Portland", state_province: "ME", postal_code: "04101", country_code: "US" },
            ship_to: { name: "Buyer", address_line1: "456 Oak Ave", city_locality: "New York", state_province: "NY", postal_code: "10001", country_code: "US" },
            packages: [{ weight: { value: 200, unit: "pound" }, dimensions: { length: 48, width: 40, height: 36, unit: "inch" } }],
          },
        }),
      });
      console.log("Rate estimate status:", rateRes.status);
      if (rateRes.ok) {
        const rateData = await rateRes.json();
        const rates = Array.isArray(rateData) ? rateData : [];
        console.log(`Rates returned: ${rates.length}`);
        for (const r of rates.slice(0, 10)) {
          console.log(`  $${r.shipping_amount?.amount || "?"} — ${r.carrier_friendly_name || "?"} ${r.service_type || ""} (${r.delivery_days || "?"} days)`);
        }
      } else {
        console.log("Rate estimate failed:", (await rateRes.text()).slice(0, 300));
      }
    }

    console.log("\n✅ Carrier discovery complete");
  } catch (e) {
    console.log("Exception:", e);
  }
}

main();
```
