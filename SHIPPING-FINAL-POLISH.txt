```typescript
/**
 * EasyPost Integration — Address Verification + Backup Carrier Rates
 *
 * EasyPost provides:
 *   1. Address verification (deliverability, residential detection)
 *   2. Backup carrier rate shopping (USPS, UPS, FedEx)
 *
 * Used as a SUPPLEMENT to Shippo — not a replacement.
 * Shippo = primary carrier rates
 * EasyPost = address verification + backup rate source if Shippo fails
 */

const EASYPOST_API_KEY = process.env.EASYPOST_TEST_API_KEY || process.env.EASYPOST_API_KEY || "";
const EASYPOST_BASE = "https://api.easypost.com/v2";

type AddressVerificationResult = {
  verified: boolean;
  residential: boolean | null;
  correctedZip: string | null;
  correctedCity: string | null;
  correctedState: string | null;
  correctedStreet: string | null;
  message: string;
};

type EasyPostRate = {
  carrier: string;
  service: string;
  rate: number;
  currency: string;
  estimatedDays: number;
  isLive: boolean;
};

// ─── Helpers ──────────────────────────────────────────────────────────────────

function authHeaders(): Record<string, string> {
  return {
    Authorization: `Basic ${Buffer.from(EASYPOST_API_KEY + ":").toString("base64")}`,
    "Content-Type": "application/json",
  };
}

function isConfigured(): boolean {
  return !!EASYPOST_API_KEY && EASYPOST_API_KEY !== "your_easypost_api_key_here";
}

// ─── Address Verification ─────────────────────────────────────────────────────

/**
 * Verify a shipping address using EasyPost.
 * Returns verification status, residential flag, and corrected fields.
 */
export async function verifyAddress(
  street1: string,
  city: string,
  state: string,
  zip: string,
  street2?: string,
  country?: string
): Promise<AddressVerificationResult> {
  if (!isConfigured()) {
    console.log("[easypost] Address verification skipped — no API key configured");
    return { verified: false, residential: null, correctedZip: null, correctedCity: null, correctedState: null, correctedStreet: null, message: "EasyPost not configured" };
  }

  try {
    const res = await fetch(`${EASYPOST_BASE}/addresses`, {
      method: "POST",
      headers: authHeaders(),
      body: JSON.stringify({
        address: {
          street1,
          street2: street2 || "",
          city,
          state,
          zip,
          country: country || "US",
          verify: ["delivery"],
        },
      }),
    });

    if (!res.ok) {
      const errText = await res.text();
      console.error("[easypost] Address verification failed:", res.status, errText.slice(0, 300));
      return { verified: false, residential: null, correctedZip: null, correctedCity: null, correctedState: null, correctedStreet: null, message: `API error: ${res.status}` };
    }

    const data = await res.json();
    const verifications = data.verifications?.delivery;
    const isVerified = verifications?.success === true;
    const isResidential = data.residential ?? null;

    console.log(`[easypost] Address verified: ${isVerified} | Residential: ${isResidential} | ${data.street1}, ${data.city} ${data.state} ${data.zip}`);

    return {
      verified: isVerified,
      residential: isResidential,
      correctedZip: data.zip || null,
      correctedCity: data.city || null,
      correctedState: data.state || null,
      correctedStreet: data.street1 || null,
      message: isVerified ? "Address verified" : (verifications?.errors?.[0]?.message || "Verification failed"),
    };
  } catch (err) {
    console.error("[easypost] Address verification exception:", err);
    return { verified: false, residential: null, correctedZip: null, correctedCity: null, correctedState: null, correctedStreet: null, message: "Exception during verification" };
  }
}

// ─── Backup Carrier Rates ─────────────────────────────────────────────────────

/**
 * Get shipping rates from EasyPost as a backup source.
 * Returns rates from all available carriers (USPS, UPS, FedEx, etc.)
 */
export async function getEasyPostRates(
  fromZip: string,
  toZip: string,
  weightLbs: number,
  lengthIn: number,
  widthIn: number,
  heightIn: number
): Promise<EasyPostRate[]> {
  if (!isConfigured()) {
    console.log("[easypost] Rate request skipped — no API key configured");
    return [];
  }

  try {
    const res = await fetch(`${EASYPOST_BASE}/shipments`, {
      method: "POST",
      headers: authHeaders(),
      body: JSON.stringify({
        shipment: {
          from_address: {
            zip: fromZip,
            country: "US",
          },
          to_address: {
            zip: toZip,
            country: "US",
          },
          parcel: {
            length: lengthIn,
            width: widthIn,
            height: heightIn,
            weight: weightLbs * 16, // EasyPost uses ounces
          },
        },
      }),
    });

    if (!res.ok) {
      const errText = await res.text();
      console.error("[easypost] Rate request failed:", res.status, errText.slice(0, 300));
      return [];
    }

    const data = await res.json();
    const rates = data.rates || [];

    const ALLOWED_CARRIERS = ["usps", "ups", "fedex", "upsdap"];

    const mapped: EasyPostRate[] = rates
      .filter((r: any) => ALLOWED_CARRIERS.includes(r.carrier?.toLowerCase()))
      .map((r: any) => ({
        carrier: r.carrier === "UPSDAP" ? "UPS" : r.carrier,
        service: r.service || "Standard",
        rate: parseFloat(r.rate) || 0,
        currency: r.currency || "USD",
        estimatedDays: r.est_delivery_days || r.delivery_days || 5,
        isLive: true,
      }))
      .filter((r: EasyPostRate) => r.rate > 0)
      .sort((a: EasyPostRate, b: EasyPostRate) => a.rate - b.rate);

    console.log(`[easypost] ${mapped.length} backup rates: ${mapped.map(r => `${r.carrier} ${r.service} $${r.rate}`).join(" | ")}`);

    return mapped;
  } catch (err) {
    console.error("[easypost] Rate request exception:", err);
    return [];
  }
}

// ─── Address Verification with Rate Optimization ──────────────────────────────

/**
 * Verify an address and return whether it's residential.
 * Useful before rate shopping — residential addresses get different rates.
 */
export async function checkResidential(zip: string): Promise<boolean | null> {
  if (!isConfigured()) return null;
  try {
    const result = await verifyAddress("", "", "", zip);
    return result.residential;
  } catch {
    return null;
  }
}

export { isConfigured as isEasyPostConfigured };
```

import "dotenv/config";

const SHIPENGINE_KEY = process.env.SHIPENGINE_API_KEY || "";
const SHIPENGINE_URL = process.env.SHIPENGINE_BASE_URL || "https://api.shipengine.com";
const FEDEX_LTL_KEY = process.env.FEDEX_LTL_API_KEY || "";
const FEDEX_LTL_SECRET = process.env.FEDEX_LTL_SECRET_KEY || "";

const FEDEX_ACCOUNT = process.env.FEDEX_ACCOUNT_NUMBER || "";

async function testShipEngine() {
  console.log("=== SHIPENGINE LTL DIAGNOSTIC ===");
  console.log("API Key:", SHIPENGINE_KEY ? `SET (${SHIPENGINE_KEY.length} chars, starts with ${SHIPENGINE_KEY.slice(0, 10)}...)` : "MISSING");
  console.log("Base URL:", SHIPENGINE_URL);

  if (!SHIPENGINE_KEY) {
    console.log("SKIP — no ShipEngine key");
    return;
  }

  // First: list connected carriers
  try {
    const carriersRes = await fetch(`${SHIPENGINE_URL}/v1/carriers`, {
      headers: { "API-Key": SHIPENGINE_KEY, "Content-Type": "application/json" },
    });
    if (carriersRes.ok) {
      const data = await carriersRes.json();
      console.log(`Connected carriers: ${data.carriers?.length || 0}`);
      for (const c of (data.carriers || [])) {
        console.log(`  - ${c.friendly_name || c.carrier_id} (${c.carrier_id})`);
      }
    } else {
      console.log("List carriers failed:", carriersRes.status, await carriersRes.text().then(t => t.slice(0, 200)));
    }
  } catch (e) {
    console.log("List carriers exception:", e);
  }

  // Second: try a rate estimate
  try {
    const rateRes = await fetch(`${SHIPENGINE_URL}/v1/rates/estimate`, {
      method: "POST",
      headers: { "API-Key": SHIPENGINE_KEY, "Content-Type": "application/json" },
      body: JSON.stringify({
        carrier_ids: [],
        from_country_code: "US",
        from_postal_code: "04901",
        to_country_code: "US",
        to_postal_code: "10001",
        weight: { value: 200, unit: "pound" },
        dimensions: { length: 48, width: 40, height: 36, unit: "inch" },
        confirmation: "none",
        ship_date: new Date().toISOString().slice(0, 10),
      }),
    });

    const rateBody = await rateRes.text();
    console.log("Rate estimate status:", rateRes.status);

    if (rateRes.ok) {
      const rates = JSON.parse(rateBody);
      console.log(`Rate estimates: ${Array.isArray(rates) ? rates.length : "N/A"}`);
      if (Array.isArray(rates)) {
        for (const r of rates.slice(0, 10)) {
          console.log(`  $${r.shipping_amount?.amount || "?"} — ${r.carrier_friendly_name || r.carrier_id} ${r.service_type || ""} (${r.delivery_days || "?"} days)`);
        }
      } else {
        console.log("Response:", JSON.stringify(rates).slice(0, 500));
      }
    } else {
      console.log("Rate estimate failed:", rateBody.slice(0, 500));
    }
  } catch (e) {
    console.log("Rate estimate exception:", e);
  }
  console.log("");
}

async function testFedExLTL() {
  console.log("=== FEDEX LTL DIAGNOSTIC ===");
  console.log("API Key:", FEDEX_LTL_KEY ? `SET (${FEDEX_LTL_KEY.length} chars)` : "MISSING");
  console.log("Secret:", FEDEX_LTL_SECRET ? `SET (${FEDEX_LTL_SECRET.length} chars)` : "MISSING");
  console.log("URL:", FEDEX_LTL_URL);
  console.log("Account:", FEDEX_ACCOUNT || "MISSING");

  if (!FEDEX_LTL_KEY || !FEDEX_LTL_SECRET) {
    console.log("SKIP — no FedEx LTL credentials");
    return;
  }

  // Auth
  let token = "";
  try {
    const authRes = await fetch(`${FEDEX_LTL_URL}/oauth/token`, {
      method: "POST",
      headers: { "Content-Type": "application/x-www-form-urlencoded" },
      body: new URLSearchParams({
        grant_type: "client_credentials",
        client_id: FEDEX_LTL_KEY,
        client_secret: FEDEX_LTL_SECRET,
      }),
    });
    if (authRes.ok) {
      const authData = await authRes.json();
      token = authData.access_token;
      console.log("Auth: SUCCESS (token length:", token.length + ")");
    } else {
      const body = await authRes.text();
      console.log("Auth: FAILED —", authRes.status, body.slice(0, 300));
      return;
    }
  } catch (e) {
    console.log("Auth exception:", e);
    return;
  }

  // LTL Rate request
  const reqBody = {
    accountNumber: { value: FEDEX_ACCOUNT },
    requestedShipment: {
      shipper: { address: { postalCode: "04901", countryCode: "US" } },
      recipient: { address: { postalCode: "10001", countryCode: "US", residential: true } },
      pickupType: "USE_SCHEDULED_PICKUP",
      rateRequestType: ["LIST", "ACCOUNT"],
      shippingChargesPayment: {
        paymentType: "SENDER",
        payor: { responsibleParty: { accountNumber: { value: FEDEX_ACCOUNT } } },
      },
      freightShipmentDetail: {
        fedExFreightBillingContactAndAddress: {
          address: { postalCode: "04901", countryCode: "US" },
        },
        lineItems: [{
          freightClass: "CLASS_100",
          weight: { units: "LB", value: 200 },
          dimensions: { length: 48, width: 40, height: 36, units: "IN" },
          pieces: 1,
          description: "General merchandise",
        }],
      },
    },
  };

  try {
    const rateRes = await fetch(`${FEDEX_LTL_URL}/rate/v1/rates/quotes`, {
      method: "POST",
      headers: {
        Authorization: `Bearer ${token}`,
        "Content-Type": "application/json",
        "X-locale": "en_US",
      },
      body: JSON.stringify(reqBody),
    });

    const rateBody = await rateRes.text();
    console.log("Rate status:", rateRes.status);

    if (rateRes.ok) {
      const rateData = JSON.parse(rateBody);
      const details = rateData.output?.rateReplyDetails || [];
      console.log(`Rate: SUCCESS — ${details.length} services returned`);
      for (const d of details) {
        const charge = d.ratedShipmentDetails?.[0]?.totalNetCharge ?? d.ratedShipmentDetails?.[0]?.totalNetFedExCharge ?? 0;
        const days = d.commit?.dateDetail?.dayCount || d.commit?.transitDays?.amount || "?";
        console.log(`  $${charge} — ${d.serviceType} (${days} days)`);
      }
    } else {
      console.log("Rate: FAILED");
      console.log("ERROR:", rateBody.slice(0, 500));
    }
  } catch (e) {
    console.log("Rate exception:", e);
  }
  console.log("");
}
