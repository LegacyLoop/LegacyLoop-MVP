```jsx
// --- Priority Action Banner ---
const [dismissedBanner, setDismissedBanner] = useState(false);

const actionItems = useMemo(() => {
  const items: { label: string; action: () => void; urgent: boolean }[] = [];

  // Items needing quotes
  const needingQuotes = shippingItems.filter(
    (item: any) => !item.quotes || item.quotes.length === 0
  ).length;
  if (needingQuotes > 0) {
    items.push({
      label: `${needingQuotes} need quotes`,
      action: () => setActiveTab("preSale"),
      urgent: needingQuotes > 5,
    });
  }

  // Items awaiting shipment > 3 days
  const overdue = shippingItems.filter((item: any) => {
    if (item.status !== "readyToShip") return false;
    const createdAt = new Date(item.createdAt);
    const daysSince = (Date.now() - createdAt.getTime()) / (1000 * 60 * 60 * 24);
    return daysSince > 3;
  }).length;
  if (overdue > 0) {
    items.push({
      label: `${overdue} awaiting shipment > 3d`,
      action: () => setActiveTab("readyToShip"),
      urgent: true,
    });
  }

  // Quotes expiring soon (< 24h from now)
  const expiring = shippingItems.filter((item: any) => {
    if (!item.quotes || item.quotes.length === 0) return false;
    return item.quotes.some((q: any) => {
      if (!q.valid_until) return false;
      const until = new Date(q.valid_until);
      const hoursUntil = (until.getTime() - Date.now()) / (1000 * 60 * 60);
      return hoursUntil > 0 && hoursUntil < 24;
    });
  }).length;
  if (expiring > 0) {
    items.push({
      label: `${expiring} quotes expiring soon`,
      action: () => setActiveTab("preSale"),
      urgent: false,
    });
  }

  // Shipments with exceptions (tracking status not "delivered")
  const exceptions = shippingItems.filter((item: any) => {
    if (item.status !== "shipped") return false;
    return item.trackingStatus && !item.trackingStatus.includes("delivered");
  }).length;
  if (exceptions > 0) {
    items.push({
      label: `${exceptions} shipments with exceptions`,
      action: () => setActiveTab("shipped"),
      urgent: true,
    });
  }

  return items;
}, [shippingItems, setActiveTab]);

// Render banner if there are action items and not dismissed
{actionItems.length > 0 && !dismissedBanner && (
  <div style={{
    display: "flex",
    flexWrap: "wrap",
    gap: "0.4rem",
    padding: "0.5rem 0.75rem",
    marginBottom: "0.75rem",
    borderRadius: "0.6rem",
    background: "rgba(255,152,0,0.04)",
    border: "1px solid rgba(255,152,0,0.12)",
    alignItems: "center",
  }}>
    <span style={{
      fontSize: "0.62rem",
      fontWeight: 700,
      color: "#ff9800",
      textTransform: "uppercase",
      letterSpacing: "0.05em",
      whiteSpace: "nowrap",
    }}>
      ⚡ Action Required
    </span>
    {actionItems.map((ai, idx) => (
      <button
        key={idx}
        onClick={ai.action}
        style={{
          fontSize: "0.6rem",
          fontWeight: 600,
          padding: "2px 8px",
          borderRadius: "9999px",
          background: ai.urgent ? "rgba(244,67,54,0.1)" : "rgba(255,152,0,0.08)",
          color: ai.urgent ? "#f44336" : "#ff9800",
          border: `1px solid ${ai.urgent ? "rgba(244,67,54,0.2)" : "rgba(255,152,0,0.15)"}`,
          cursor: "pointer",
          transition: "all 0.2s ease",
          whiteSpace: "nowrap",
        }}
        onMouseEnter={(e) => {
          const el = e.currentTarget;
          el.style.opacity = "0.8";
        }}
        onMouseLeave={(e) => {
          const el = e.currentTarget;
          el.style.opacity = "1";
        }}
      >
        {ai.label}
      </button>
    ))}
    <button
      onClick={() => setDismissedBanner(true)}
      style={{

        color: "var(--text-muted)",

        cursor: "pointer",
        padding: "0 0.25rem",
        whiteSpace: "nowrap",
      }}
    >
      ✕
    </button>
  </div>
)}
```

```typescript
import { NextRequest, NextResponse } from "next/server";
import { auth } from "@clerk/nextjs/server";
import { prisma } from "@/lib/db";

export async function POST(req: NextRequest) {
  try {
    const { userId } = await auth();
    if (!userId) {
      return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
    }

    const { itemIds } = await req.json() as { itemIds: string[] };
    if (!itemIds || !Array.isArray(itemIds) || itemIds.length === 0) {
      return NextResponse.json(
        { error: "Invalid itemIds array" },
        { status: 400 }
      );
    }

    // Fetch pickup status for all items in parallel
    const items = await prisma.item.findMany({
      where: {
        id: { in: itemIds },
        userId,
      },
      select: {
        id: true,
        localPickupStatus: true,
        localPickupScheduledDate: true,
        localPickupContactName: true,
        localPickupContactPhone: true,
      },
    });

    // Build response: statuses Record<itemId, PickupStatus>
    const statuses: Record<string, any> = {};
    for (const item of items) {
      statuses[item.id] = {
        itemId: item.id,
        status: item.localPickupStatus || "pending",
        scheduledDate: item.localPickupScheduledDate,
        contactName: item.localPickupContactName,
        contactPhone: item.localPickupContactPhone,
        lastUpdated: new Date().toISOString(),
      };
    }

    console.log(`[pickup-batch] Returned statuses for ${items.length} items`);

    return NextResponse.json({
      statuses,
      count: items.length,
    });
  } catch (error) {
    console.error("[pickup-batch] Error:", error);
    return NextResponse.json(
      { error: "Internal server error" },
      { status: 500 }
    );
  }
}
```
