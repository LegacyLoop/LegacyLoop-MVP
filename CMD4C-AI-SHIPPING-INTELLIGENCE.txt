AI SHIPPING INTELLIGENCE GAP ANALYSIS:

The AI generates 4 shipping-relevant fields during item analysis:

1. weight_estimate_lbs (number | null)
   - AI's best guess at item weight in pounds
   - Currently: stored in rawJson, used as FALLBACK in estimate API
   - Problem: never saved to Item.shippingWeight, so every request must re-parse rawJson
   - Fix: Persist to new Item.aiWeightLbs field

2. shipping_difficulty (string | null) — "Easy", "Moderate", "Difficult", "Freight only"
   - AI's assessment of how hard this item is to ship
   - Currently: COMPLETELY UNUSED. Stored in rawJson, never read, never displayed
   - Problem: This is the most valuable AI shipping signal and we throw it away
   - Fix: Persist to new Item.aiShippingDifficulty field, use for fragility, method routing, UI badges

3. shipping_notes (string | null)
   - Specific packing and shipping advice from AI
   - Examples: "Ship on edge", "Drain fuel before shipping", "Wrap each piece individually"
   - Currently: COMPLETELY UNUSED in UI. Stored in rawJson, sometimes passed to suggestPackage but never displayed
   - Problem: This is personalized packing intelligence that goes to waste
   - Fix: Persist to new Item.aiShippingNotes field, display prominently in ShippingPanel

4. dimensions_estimate (string)
   - AI's dimension estimate as text: "34 x 24 x 30"
   - Currently: Stored in rawJson, parsed on-demand by parseDimensions()
   - Problem: Re-parsed every request, never split into L/W/H fields
   - Fix: Parse once in analyze route, persist to Item.aiDimsEstimate as text + parsed into shippingLength/Width/Height if empty

  // ─── Shipping Intelligence ───────────────────────────────────────
  const shippingUpdates: Record<string, any> = {};

  // Weight estimate
  const aiWeight = analysis.weight_estimate_lbs;
  if (aiWeight && typeof aiWeight === "number" && aiWeight > 0) {
    shippingUpdates.aiWeightLbs = Math.round(aiWeight * 10) / 10;
    // Also populate shippingWeight if user hasn't set it
    const existing = await prisma.item.findUnique({ where: { id: itemId }, select: { shippingWeight: true } });
    if (!existing?.shippingWeight) {
      shippingUpdates.shippingWeight = shippingUpdates.aiWeightLbs;
    }
  }

  // Dimensions estimate
  const aiDims = analysis.dimensions_estimate;
  if (aiDims && typeof aiDims === "string") {
    shippingUpdates.aiDimsEstimate = aiDims;
    // Parse into L/W/H if user hasn't set them
    const dimMatch = aiDims.match(/(\d+(?:\.\d+)?)\s*[x×X]\s*(\d+(?:\.\d+)?)\s*[x×X]\s*(\d+(?:\.\d+)?)/);
    if (dimMatch) {
      const existing = await prisma.item.findUnique({
        where: { id: itemId },
        select: { shippingLength: true, shippingWidth: true, shippingHeight: true },
      });
      if (!existing?.shippingLength) shippingUpdates.shippingLength = parseFloat(dimMatch[1]);
      if (!existing?.shippingWidth) shippingUpdates.shippingWidth = parseFloat(dimMatch[2]);
      if (!existing?.shippingHeight) shippingUpdates.shippingHeight = parseFloat(dimMatch[3]);
    }
  }

  // Shipping difficulty
  const aiDifficulty = analysis.shipping_difficulty;
  if (aiDifficulty && typeof aiDifficulty === "string") {
    shippingUpdates.aiShippingDifficulty = aiDifficulty;
    // Auto-set isFragile if AI says Difficult and not already set
    if (aiDifficulty === "Difficult") {
      const existing = await prisma.item.findUnique({ where: { id: itemId }, select: { isFragile: true } });
      if (!existing?.isFragile) shippingUpdates.isFragile = true;
    }
  }

  // Shipping notes
  const aiNotes = analysis.shipping_notes;
  if (aiNotes && typeof aiNotes === "string" && aiNotes.length > 3) {
    shippingUpdates.aiShippingNotes = aiNotes;
  }

  // Confidence
  const confidence = analysis.confidence;
  if (confidence && typeof confidence === "number") {
    shippingUpdates.aiShippingConfidence = Math.round(confidence * 100) / 100;
  }

  // Persist shipping intelligence
  if (Object.keys(shippingUpdates).length > 0) {
    await prisma.item.update({
      where: { id: itemId },
      data: shippingUpdates,
    });
    console.log(`[shipping-intel] Persisted for ${itemId}:`,
      Object.entries(shippingUpdates).map(([k, v]) => `${k}=${v}`).join(", ")
    );
  }
