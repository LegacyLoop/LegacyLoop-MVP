/**
 * Arta API Client — White-Glove Shipping for High-Value Items

const ARTA_BASE = process.env.ARTA_API_BASE_URL || "https://api.arta.io";
const ARTA_KEY = process.env.ARTA_API_KEY || "";
const ARTA_VERSION = process.env.ARTA_API_VERSION || "2021-01-01";
const ARTA_TIMEOUT = Number(process.env.ARTA_QUOTE_TIMEOUT_MS) || 6000;

// ─── Types ──────────────────────────────────────────────────────

export interface ArtaObject {

  subtype: string;
  /** Width in inches */
  width: number;
  /** Height in inches */
  height: number;
  /** Depth in inches */
  depth: number;
  /** Weight in pounds */
  weight: number;
  /** Weight unit */
  weight_unit: string;
  /** Estimated value in USD (string per Arta API) */
  value: string;
  /** Currency */
  value_currency: string;
  /** Unit of measurement for dimensions */
  unit_of_measurement: string;
  /** Current packing state */
  current_packing?: string[];
  /** Item details */
  details?: {
    title?: string;
    is_fragile?: boolean;
    notes?: string;
    creator?: string;
    creation_date?: string;
    materials?: string[];
  };
}

export interface ArtaLocation {
  /** Address line */
  address_line_1?: string;
  /** City */
  city?: string;
  /** Region/state (2-letter code) */
  region?: string;
  /** Postal code */
  postal_code: string;
  /** Country (2-letter ISO) */
  country: string;
  /** Contact info */
  contacts?: { name: string; email_address?: string; phone_number?: string }[];
}

export interface ArtaQuoteRequest {
  origin: ArtaLocation;
  destination: ArtaLocation;
  objects: ArtaObject[];
  /** Internal reference */
  internal_reference?: string;
  /** Whether to include insurance quote */
  insurance?: string; // "arta_transit_insurance"
  /** Preferred quote types to return */

  /** Quoting strategy */
  quoting_strategy?: string; // "best_rate" | "all"

export interface ArtaServiceOption {
  quote_type: string; // "premium" | "select" | "parcel"
  total: number;

  included_services: string[];

    amount: number;

  detail?: string;

  expires_at?: string;

export interface ArtaError {
  errors: { detail: string }[];

// ─── API Helpers ────────────────────────────────────────────────

async function artaFetch<T>(
  endpoint: string,
  options: { method?: string; body?: unknown } = {}
): Promise<{ data: T | null; error: string | null; status: number }> {
  const { method = "GET", body } = options;

  if (!ARTA_KEY) {
    console.error("[arta] No ARTA_API_KEY configured");
    return { data: null, error: "Arta API key not configured", status: 0 };
  }

  const controller = new AbortController();
  const timeout = setTimeout(() => controller.abort(), ARTA_TIMEOUT);

  const startTime = Date.now();

  try {
    const res = await fetch(`${ARTA_BASE}${endpoint}`, {
      method,
      headers: {
        "Authorization": `ARTA_APIKey ${ARTA_KEY}`,
        "Content-Type": "application/json",
        "Arta-API-Version": ARTA_VERSION,
        "Arta-Quote-Timeout": String(ARTA_TIMEOUT),
      },
      body: body ? JSON.stringify(body) : undefined,
      signal: controller.signal,
    });

    const elapsed = Date.now() - startTime;
    const json = await res.json();

    console.log(`[arta] ${method} ${endpoint} → ${res.status} (${elapsed}ms)`);

    if (!res.ok) {
      const errMsg = (json as ArtaError)?.errors?.[0]?.detail || `HTTP ${res.status}`;
      console.error(`[arta] Error: ${errMsg}`, JSON.stringify(json).slice(0, 500));
      return { data: null, error: errMsg, status: res.status };
    }

    return { data: json as T, error: null, status: res.status };
  } catch (err: unknown) {
    const elapsed = Date.now() - startTime;
    if ((err as Error).name === "AbortError") {
      console.error(`[arta] Request timed out after ${ARTA_TIMEOUT}ms`);
      return { data: null, error: `Arta request timed out (${ARTA_TIMEOUT}ms)`, status: 0 };
    }
    console.error(`[arta] Fetch error (${elapsed}ms):`, (err as Error).message);
    return { data: null, error: (err as Error).message, status: 0 };
  } finally {
    clearTimeout(timeout);
  }
}

// ─── Public API ─────────────────────────────────────────────────

/**
 * Request quotes from Arta for a high-value item.
 *
 * Returns multiple service tiers: premium (white glove), select, parcel.
 * Each tier includes pricing, insurance options, and service details.
 */
export async function getArtaQuote(
  request: ArtaQuoteRequest
): Promise<{ quotes: ArtaServiceOption[]; requestId: string | null; error: string | null; hosted_session_url?: string }> {
  console.log(`[arta] Requesting quote for ${request.objects.length} object(s)`,
    `${request.origin.postal_code} → ${request.destination.postal_code}`);

  const { data, error, status } = await artaFetch<ArtaQuoteResponse>("/requests", {
    method: "POST",
    body: { request },
  });

  if (error || !data) {
    return { quotes: [], requestId: null, error: error || "No response from Arta" };
  }

  console.log(`[arta] Quote ${data.id}: ${data.status}, ${data.quotes?.length || 0} options`,
    data.quotes?.map(q => `${q.quote_type}=$${q.total}`).join(", "));

  return {
    quotes: data.quotes || [],
    requestId: data.id,
    error: null,
    hosted_session_url: data.hosted_session_url,
  };
}

/**
 * Get an existing quote request by ID.
 */
export async function getArtaRequest(requestId: string): Promise<{
  data: ArtaQuoteResponse | null;
  error: string | null;
}> {
  const { data, error } = await artaFetch<ArtaQuoteResponse>(`/requests/${requestId}`);
  return { data, error };
}

/**
 * Check if an item qualifies for Arta white-glove shipping.
 *
 * Eligibility criteria:
 * - Premium items (value > $2,000)
 * - High-value antiques (value > $500 AND is antique)
 * - High-value fragile items (value > $500 AND is fragile)
 * - Difficult shipping items (value > $500 AND AI says Difficult/Freight only)
 * - Art, Antique, or Collectible categories
 */
export function isArtaEligible(item: {
  valueLow?: number | null;
  valueHigh?: number | null;
  isFragile?: boolean;
  aiShippingDifficulty?: string | null;
  category?: string | null;
  isAntique?: boolean;
}): boolean {
  const valMid = ((item.valueLow || 0) + (item.valueHigh || 0)) / 2;
  const isPremium = valMid > 2000;
  const isHighValue = valMid > 500;

  // Premium items always qualify
  if (isPremium) return true;

  // High-value antiques
  if (isHighValue && item.isAntique) return true;

  // High-value fragile items
  if (isHighValue && item.isFragile) return true;

  // High-value difficult shipping
  if (isHighValue && (item.aiShippingDifficulty === "Difficult" || item.aiShippingDifficulty === "Freight only")) return true;

  // Art, antique, or collectible categories
  const artCategories = ["art", "antique", "collectible", "fine art", "sculpture", "painting"];
  if (isHighValue && item.category && artCategories.includes(item.category.toLowerCase())) return true;

  return false;
}

/**
 * Build an Arta quote request from LegacyLoop item data.
 */
export function buildArtaRequest(item: {
  title?: string;
  category?: string | null;
  aiWeightLbs?: number | null;
  shippingWeight?: number | null;
  aiDimsEstimate?: string | null;
  shippingLength?: number | null;
  shippingWidth?: number | null;
  shippingHeight?: number | null;
  valueLow?: number | null;
  valueHigh?: number | null;
  isFragile?: boolean;
  aiShippingNotes?: string | null;
  id?: string;
}, origin: { zip: string }, destination: { zip: string }): ArtaQuoteRequest {
  // Weight: prefer user-entered, fall back to AI estimate
  const weight = item.shippingWeight || item.aiWeightLbs || 20;

  // Dimensions: prefer user-entered, fall back to AI, then defaults
  let length = item.shippingLength || 24;
  let width = item.shippingWidth || 18;
  let height = item.shippingHeight || 18;

  if (!item.shippingLength && item.aiDimsEstimate) {
    const match = item.aiDimsEstimate.match(/(\d+(?:\.\d+)?)\s*[x×X]\s*(\d+(?:\.\d+)?)\s*[x×X]\s*(\d+(?:\.\d+)?)/);
    if (match) {
      length = parseFloat(match[1]);
      width = parseFloat(match[2]);
      height = parseFloat(match[3]);
    }
  }

  const valMid = ((item.valueLow || 0) + (item.valueHigh || 0)) / 2;

  return {
    origin: {
      postal_code: origin.zip,
      country: "US",
    },
    destination: {
      postal_code: destination.zip,
      country: "US",
    },
    objects: [{
      subtype: item.title || item.category || "Antique / Collectible",
      width,
      height,
      depth: length,
      weight,
      weight_unit: "lb",
      value: String(Math.round(valMid * 100) / 100),
      value_currency: "USD",
      unit_of_measurement: "in",
      current_packing: ["no_packing"],
      details: {
        title: item.title || "Item",
        is_fragile: item.isFragile || false,
        notes: item.aiShippingNotes || undefined,
      },
    }],
    internal_reference: item.id || undefined,
    insurance: "arta_transit_insurance",
    preferred_quote_types: ["premium", "select", "parcel"],
    quoting_strategy: "best_rate",
    shipping_notes: item.aiShippingNotes || "LegacyLoop item — handle with care",
  };
}

/**
 * POST /api/shipping/arta-quote
 * Get white-glove shipping quotes from Arta for qualifying items.
 *
 * Body: { item: { title, category, valueLow, valueHigh, shippingWeight, aiWeightLbs,
 *          shippingLength, shippingWidth, shippingHeight, aiDimsEstimate,
 *          isFragile, aiShippingDifficulty, isAntique, id },
 *         fromZip, toZip }
 */
export async function POST(req: NextRequest) {
  const user = await authAdapter.getSession();
  if (!user) return NextResponse.json({ error: "Unauthorized" }, { status: 401 });

  try {
    const body = await req.json();
    const { item, fromZip, toZip } = body;

    if (!toZip) {
      return NextResponse.json({ error: "Destination zip required" }, { status: 400 });
    }

    // Check eligibility
    if (!isArtaEligible(item)) {
      console.log(`[arta-quote] Item ${item?.id} not eligible for Arta`);
      return NextResponse.json({
        eligible: false,
        quotes: [],
        message: "Item does not meet Arta white-glove eligibility criteria",
      });
    }

    // Build and send request
    const artaReq = buildArtaRequest(
      item,
      { zip: fromZip || "04901" },
      { zip: toZip }
    );

    console.log(`[arta-quote] Requesting for item ${item?.id}:`,
      `$${artaReq.objects[0]?.value} value,`,
      `${artaReq.objects[0]?.weight}lbs,`,
      `${fromZip || "04901"} → ${toZip}`);

    const result = await getArtaQuote(artaReq);

    if (result.error) {
      console.error(`[arta-quote] Error: ${result.error}`);
      return NextResponse.json({
        eligible: true,
        quotes: [],
        error: result.error,
        requestId: result.requestId,
        source: "arta",
        isLive: false,
      });
    }

    // Format quotes for frontend
    const formattedQuotes = result.quotes.map(q => ({
      quote_type: q.quote_type,
      total: q.total,
      currency: q.total_currency || "USD",
      services: q.included_services || [],
      insurance: q.included_insurance_policy || null,
      is_estimated: q.is_estimated,
      detail: q.detail || null,
      source: "arta",
      isLive: true,
    }));

    console.log(`[arta-quote] Success: ${formattedQuotes.length} options`,
      formattedQuotes.map(q => `${q.quote_type}=$${q.total}`).join(", "));

    return NextResponse.json({
      eligible: true,
      quotes: formattedQuotes,
      requestId: result.requestId,
      hosted_session_url: result.hosted_session_url || null,
      source: "arta",
      isLive: true,
      artaMode: process.env.ARTA_MODE || "test",
    });
  } catch (err: unknown) {
    console.error("[arta-quote] Unhandled error:", (err as Error).message);
    return NextResponse.json(
      { error: "Failed to get Arta quote", detail: (err as Error).message },
      { status: 500 }
    );
  }
}

import { NextRequest, NextResponse } from "next/server";

/**
 * POST /api/webhooks/arta
 * Receives webhook events from Arta.
 *
 * Events: quote.ready, shipment.created, shipment.collected,
 *         shipment.in_transit, shipment.delivered, shipment.exception
 *
 * For now: log all events. Future: update shipment status in DB.
 */
export async function POST(req: NextRequest) {
  try {
    const body = await req.json();
    const event = body?.event_type || body?.type || "unknown";
    const resourceId = body?.resource?.id || body?.data?.id || "unknown";

    console.log(`[arta-webhook] Event: ${event}`, JSON.stringify({
      resource_id: resourceId,
      status: body?.resource?.status || body?.data?.status,
      timestamp: new Date().toISOString(),
    }));

    // TODO: When Arta goes production, update shipment tracking in DB
    // For now, log and acknowledge
    switch (event) {
      case "quote.ready":
        console.log(`[arta-webhook] Quote ready: ${resourceId}`);
        break;
      case "shipment.created":
        console.log(`[arta-webhook] Shipment created: ${resourceId}`);
        break;
      case "shipment.collected":
        console.log(`[arta-webhook] Shipment collected: ${resourceId}`);
        break;
      case "shipment.in_transit":
        console.log(`[arta-webhook] Shipment in transit: ${resourceId}`);
        break;
      case "shipment.delivered":
        console.log(`[arta-webhook] Shipment delivered: ${resourceId}`);
        break;
      case "shipment.exception":
        console.error(`[arta-webhook] Shipment exception: ${resourceId}`, body);
        break;
      default:
        console.log(`[arta-webhook] Unhandled event: ${event}`);
    }

    return NextResponse.json({ received: true });
  } catch (err: unknown) {
    console.error("[arta-webhook] Parse error:", (err as Error).message);
    return NextResponse.json({ received: true }, { status: 200 }); // Always 200 to Arta
  }
}
