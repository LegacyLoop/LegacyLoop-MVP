STEP 1 — CREATE lib/shipping/shipengine-ltl.ts (NEW FILE):

/**
 * ShipEngine LTL API Integration
 * Provides real LTL freight quotes, pickup scheduling,
 * BOL generation, and tracking via ShipEngine/ShipStation API.
 *
 * Sandbox carriers: test, warn, fail
 * Production: connect real carriers (XPO, Old Dominion, etc.)
 */

const SHIPENGINE_KEY = process.env.SHIPENGINE_API_KEY || "";
const SHIPENGINE_BASE = process.env.SHIPENGINE_BASE_URL || "https://api.shipengine.com";

interface LTLAddress {
  company_name: string;
  address_line1: string;
  city_locality: string;
  state_province: string;
  postal_code: string;
  country_code?: string;
  contact_name?: string;
  phone?: string;
  email?: string;
}

interface LTLPackage {
  weight: { value: number; unit: "pound" };
  dimensions: { length: number; width: number; height: number; unit: "inch" };
  freight_class: string;
  description?: string;
  nmfc_code?: string;
}

interface LTLQuoteRequest {
  carrier_id: string;
  ship_from: LTLAddress;
  ship_to: LTLAddress;
  packages: LTLPackage[];
  accessorials?: string[];
}

interface LTLQuoteResponse {
  quote_id: string;
  carrier: string;
  service: string;
  total_amount: number;
  currency: string;
  transit_days: number;
  valid_until: string;
  charges: { description: string; amount: number }[];
}

// Connect sandbox test carriers (one-time)
export async function connectTestCarriers(): Promise<string[]> {
  const carriers = ["test", "warn", "fail"];
  const ids: string[] = [];
  for (const c of carriers) {
    try {
      const res = await fetch(`${SHIPENGINE_BASE}/v-beta/ltl/connections/${c}`, {
        method: "POST",
        headers: { "API-Key": SHIPENGINE_KEY, "Content-Type": "application/json" },
        body: JSON.stringify({ credentials: {} }),
      });
      const data = await res.json();
      if (data.carrier_id) ids.push(data.carrier_id);
    } catch {}
  }
  return ids;
}

// Get LTL quote from ShipEngine
export async function getLTLQuote(req: LTLQuoteRequest): Promise<LTLQuoteResponse | null> {
  if (!SHIPENGINE_KEY) return null;
  try {
    const res = await fetch(`${SHIPENGINE_BASE}/v-beta/ltl/quotes`, {
      method: "POST",
      headers: { "API-Key": SHIPENGINE_KEY, "Content-Type": "application/json" },
      body: JSON.stringify(req),
    });
    if (!res.ok) {
      console.error("[ShipEngine LTL] Quote error:", res.status, await res.text());
      return null;
    }
    return await res.json();
  } catch (e) {
    console.error("[ShipEngine LTL] Quote failed:", e);
    return null;
  }
}

// Schedule LTL pickup
export async function scheduleLTLPickup(quoteId: string, pickupDate: string): Promise<any> {
  if (!SHIPENGINE_KEY) return null;
  try {
    const res = await fetch(`${SHIPENGINE_BASE}/v-beta/ltl/pickups`, {
      method: "POST",
      headers: { "API-Key": SHIPENGINE_KEY, "Content-Type": "application/json" },
      body: JSON.stringify({ quote_id: quoteId, pickup_date: pickupDate }),
    });
    return await res.json();
  } catch (e) {
    console.error("[ShipEngine LTL] Pickup failed:", e);
    return null;
  }
}

// Get LTL documents (BOL, invoice, proof of delivery)
export async function getLTLDocuments(pickupId: string): Promise<any> {
  if (!SHIPENGINE_KEY) return null;
  try {
    const res = await fetch(`${SHIPENGINE_BASE}/v-beta/ltl/pickups/${pickupId}/documents`, {
      headers: { "API-Key": SHIPENGINE_KEY },
    });
    return await res.json();
  } catch (e) {
    console.error("[ShipEngine LTL] Documents failed:", e);
    return null;
  }
}

export type { LTLAddress, LTLPackage, LTLQuoteRequest, LTLQuoteResponse };

STEP 2 — CREATE app/api/shipping/ltl-quote/route.ts (NEW FILE):

This API route receives freight quote requests from the
Shipping Center and calls ShipEngine:

import { NextRequest, NextResponse } from "next/server";
import { authAdapter } from "@/lib/adapters/auth";
import { getLTLQuote, connectTestCarriers } from "@/lib/shipping/shipengine-ltl";
import type { LTLQuoteRequest } from "@/lib/shipping/shipengine-ltl";

// Cache carrier IDs (they don't change per session)
let cachedCarrierIds: string[] = [];

export async function POST(req: NextRequest) {
  const user = await authAdapter.getSession();
  if (!user) return NextResponse.json({ error: "Unauthorized" }, { status: 401 });

  const body = await req.json();
  const { fromZip, toZip, weight, length, width, height, freightClass, description } = body;

  // Connect test carriers if not yet connected
  if (cachedCarrierIds.length === 0) {
    cachedCarrierIds = await connectTestCarriers();
  }

  if (cachedCarrierIds.length === 0) {
    return NextResponse.json({ error: "No LTL carriers connected", quotes: [] });
  }

  // Get quotes from all connected carriers
  const quotes = [];
  for (const carrierId of cachedCarrierIds) {
    const quoteReq: LTLQuoteRequest = {
      carrier_id: carrierId,
      ship_from: {
        company_name: "LegacyLoop Seller",
        address_line1: "123 Main St",
        city_locality: "Portland",
        state_province: "ME",
        postal_code: fromZip || "04101",
        country_code: "US",
      },
      ship_to: {
        company_name: "Buyer",
        address_line1: "456 Destination St",
        city_locality: "Destination",
        state_province: "NY",
        postal_code: toZip || "10001",
        country_code: "US",
      },
      packages: [{
        weight: { value: Number(weight) || 100, unit: "pound" },
        dimensions: {
          length: Number(length) || 48,
          width: Number(width) || 40,
          height: Number(height) || 36,
          unit: "inch",
        },
        freight_class: freightClass || "70",
        description: description || "Household goods",
      }],
    };

    const quote = await getLTLQuote(quoteReq);
    if (quote) quotes.push(quote);
  }

  return NextResponse.json({ quotes, carrierCount: cachedCarrierIds.length });
}
