    function AccordionHeader({
      id, icon, title, subtitle, isOpen, onToggle, accentColor, badge,
    }: {

      isOpen: boolean; onToggle: (id: string) => void;

            if (!isOpen) (e.currentTarget as HTMLButtonElement).style.background = "none";
