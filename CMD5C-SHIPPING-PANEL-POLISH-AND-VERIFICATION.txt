// Compute Arta eligibility for each item
const isArtaEligible = (item: any) => {
  const isPremium = item.isPremium === true;
  const isHighValue = item.value && item.value > 500;
  const isAntique = item.isAntique === true;
  const isFragile = item.isFragile === true;
  const isDifficult = item.aiShippingDifficulty === "Difficult" || item.aiShippingDifficulty === "Freight only";

  return isPremium || (isHighValue && isAntique) || (isHighValue && isFragile) || (isHighValue && isDifficult);
};

```typescript
import { NextRequest, NextResponse } from "next/server";

export async function POST(req: NextRequest) {
  try {
    const body = await req.json();
    const { itemId, weight, length, width, height, fromZip, toZip } = body;

    if (!toZip || !weight) {
      return NextResponse.json(
        { error: "Missing toZip or weight" },
        { status: 400 }
      );
    }

    console.log(`[arta] Quote request: item=${itemId}, weight=${weight}lbs, from=${fromZip}, to=${toZip}`);

    // Placeholder: Call actual Arta API here
    // For now, return mock quote tiers matching Arta's actual pricing structure
    const quotes = {
      premium: {
        tier: "Premium",
        description: "White-glove service with dedicated concierge",
        basePrice: 250,
        transitDays: "2-3",
        isWhiteGlove: true,
      },
      select: {
        tier: "Select",
        description: "Premium service with signature required",
        basePrice: 150,
        transitDays: "3-5",
        isWhiteGlove: true,
      },
      parcel: {
        tier: "Parcel",
        description: "Standard shipping with insurance",
        basePrice: 75,
        transitDays: "5-7",
        isWhiteGlove: false,
      },
    };

    const estimatedTotal = (
      quotes.premium.basePrice +
      (weight > 50 ? (weight - 50) * 0.5 : 0) +
      (Math.abs(Number(toZip) - Number(fromZip)) > 1000 ? 50 : 0)
    );

    console.log(`[arta] Quote response: ${estimatedTotal} estimate, 3 tiers`);

    return NextResponse.json({
      tiers: quotes,
      estimatedTotal: Math.round(estimatedTotal * 100) / 100,
      source: "arta",
      isLive: false, // Set to true when real Arta API is integrated
      powered_by: "Arta",
    });
  } catch (error) {
    console.error("[arta] Error:", error);
    return NextResponse.json(
      { error: "Failed to get Arta quote" },
      { status: 500 }
    );
  }
}
```

This endpoint returns three quote tiers (Premium, Select, Parcel) that match
Arta's actual service levels. Update the basePrice and formula based on real
Arta pricing when integrating their actual API.

```jsx
        height: Number(formState.height) || 36,
        fromZip: formState.fromZip || "04901",
        toZip: formState.toZip,
      }),
    });
    const data = await res.json();
    if (res.ok) {
      console.log("[arta] Quote received:", data);
      setArtaQuoteData(data);
    } else {
      setArtaQuoteError(data.error || "Failed to fetch Arta quote");
    }
  } catch (err) {
    console.error("[arta] Error:", err);
    setArtaQuoteError("Failed to fetch Arta quote");
  } finally {
    setLoadingArtaQuote(false);
  }
};
```

This wires Arta quotes into the Shipping Center's FreightTab.

    // use data

NOT in component body:
  const data = localStorage.getItem(...); // WRONG!
