UPGRADE: For each item in the preSale array, call the

This means calling /api/shipping/rates internally (or
calling getShippingRates from lib/shipping/shippo.ts
directly) for each metro city:

  const metroZips = [
    { city: "New York, NY", zip: "10001" },
    { city: "Los Angeles, CA", zip: "90001" },
    { city: "Chicago, IL", zip: "60601" },
    { city: "Houston, TX", zip: "77001" },
  ];

  const metroEstimates = await Promise.all(
    metroZips.map(async (metro) => {
      try {
        const result = await getShippingRates(
          { zip: item.saleZip || "04101" },
          { zip: metro.zip },
          { length: pkgL, width: pkgW, height: pkgH, weight: actualWeight }
        );
        const cheapest = result.rates.sort((a, b) => a.rate - b.rate)[0];
        const fastest = result.rates.sort((a, b) => a.estimatedDays - b.estimatedDays)[0];
        return {
          city: metro.city,
          zip: metro.zip,
          estimatedCost: cheapest?.rate ?? 15,
          estimatedDays: cheapest?.estimatedDays ?? 5,
          isCheapest: false,
          isFastest: false,
          carrier: cheapest?.carrier ?? "USPS",
          service: cheapest?.service ?? "Ground",
        };
      } catch {
        return null;
      }
    })
  );
  const validMetros = metroEstimates.filter(Boolean);
  // Mark cheapest/fastest
  if (validMetros.length > 0) {
    const minCost = Math.min(...validMetros.map(m => m.estimatedCost));
    const minDays = Math.min(...validMetros.map(m => m.estimatedDays));
    validMetros.forEach(m => {
      m.isCheapest = m.estimatedCost === minCost;
      m.isFastest = m.estimatedDays === minDays;
    });
  }
